CREATE TABLE buildings(
	building_id VARCHAR(100),
	building_name VARCHAR(100));
	
CREATE TABLE classes(
	class_name VARCHAR(100),
	class_time DATETIME,
	class_room VARCHAR(100),
	building_id VARCHAR(100));

CREATE TABLE events(
	event_name VARCHAR(100),
	event_time DATETIME,
	event_room VARCHAR(100),
	building_id VARCHAR(100));
